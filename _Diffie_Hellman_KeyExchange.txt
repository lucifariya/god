

    import random

    def isPrime(n):
        for i in range(2, int(math.sqrt(n)) + 1):
            if n % i == 0:
                return False
        return True

    def getPrimeNumber(r1, r2):
        while True:
            n = random.randint(r1, r2)
            if isPrime(n):
                return n
