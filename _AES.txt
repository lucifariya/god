    from Crypto.Cipher import AES
    import binascii

    def encryption(aes, P, key):
        while (len(P) % 16 != 0):
            P += " "
        P = P.encode()
        C = aes.encrypt(P)
        C = binascii.hexlify(C)
        C = C.decode()
        return C

    def decryption(aes, C, key):
        C = binascii.unhexlify(C)
        P1 = aes.decrypt(C)
        P1 = P1.decode().strip()
        return P1
